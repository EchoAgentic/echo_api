def get_free_tier_model_and_increment():
    """
    Gère les quotas du Plan Free sur 24h avec Supabase de manière autonome.
    Basceuille entre 3.1 Flash-Lite, 2.5 Flash-Lite et 2.5 Flash selon l'utilisation.
    """
    try:
        current_date_str = datetime.now().strftime("%Y-%m-%d")
        
        # 1. Lire l'état actuel du compteur global dans Supabase
        res = supabase.table("global_api_quotas").select("*").eq("id", "free_tier_counter").single().execute()
        quota_data = res.data
        
        if not quota_data:
            return "gemini-3.1-flash-lite"
            
        count = quota_data.get("request_count", 0)
        last_reset = quota_data.get("last_reset_date")
        
        # 2. Vérifier si on a changé de journée (Réinitialisation automatique des 24h)
        if last_reset != current_date_str:
            count = 0
            supabase.table("global_api_quotas").update({
                "request_count": 0,
                "last_reset_date": current_date_str,
                "updated_at": datetime.now().isoformat()
            }).eq("id", "free_tier_counter").execute()
            print(f"[QUOTA] Nouvelle journée détectée ({current_date_str}). Remise à zéro du compteur.")

        # 3. Choix du modèle selon la stratégie de failover de Nicholas
        if count < 950:
            selected_model = "gemini-3.1-flash-lite"
        elif count < 1900:  
            selected_model = "gemini-2.5-flash-lite"
        else:
            selected_model = "gemini-2.5-flash"
            
        # 4. Incrémenter le compteur global dans Supabase (+1)
        new_count = count + 1
        supabase.table("global_api_quotas").update({
            "request_count": new_count,
            "updated_at": datetime.now().isoformat()
        }).eq("id", "free_tier_counter").execute()
        
        print(f"[QUOTA] Requête Free #{new_count} acheminée vers : {selected_model}")
        return selected_model

    except Exception as err:
        print(f"Erreur lors de la gestion des quotas Supabase : {err}")
        return "gemini-3.1-flash-lite"
