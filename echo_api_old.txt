def get_failover_count():
    try:
        if os.path.exists(COUNTER_FILE):
            with open(COUNTER_FILE, "r") as f:
                data = json.load(f)
                return data.get("count", 0)
    except Exception as e:
        print(f"[WARN] Erreur lecture compteur failover : {e}")
    return 0

def increment_failover_count():
    try:
        current = get_failover_count()
        with open(COUNTER_FILE, "w") as f:
            json.dump({"count": current + 1}, f)
        print(f"⚠️ [FAILOVER FREE] Compteur de secours utilisé : {current + 1} / {MAX_FREE_FAILOVERS}")
    except Exception as e:
        print(f"[WARN] Erreur écriture compteur failover : {e}")

def clean_and_parse_json(raw_text):
    text = raw_text.strip()

    # Retrait propre des blocs markdown JSON
    if text.startswith("```json"):
        text = text[7:]
    elif text.startswith("```"):
        text = text[3:]
    if text.endswith("```"):
        text = text[:-3]
    text = text.strip()

    try:
        parsed = json.loads(text)
        if isinstance(parsed, dict) and "response" in parsed:
            return parsed
    except Exception:
        pass

    # Essai de recherche de structures d'arrays ou d'objets JSON isolés
    match_array = re.search(r'\[\s*(\{.*?\})\s*,?', text, re.DOTALL)
    if match_array:
        try:
            parsed = json.loads(match_array.group(1))
            if isinstance(parsed, dict) and "response" in parsed:
                return parsed
        except Exception:
            pass

    match = re.search(r'\{.*\}', text, re.DOTALL)
    if match:
        try:
            parsed = json.loads(match.group(0))
            if isinstance(parsed, dict) and "response" in parsed:
                return parsed
        except Exception:
            pass

    if text:
        clean_response = text
        if '"response":' in text:
            res_match = re.search(r'"response"\s*:\s*"([^"]+)"', text)
            if res_match:
                clean_response = res_match.group(1)
        return {"action": None, "response": clean_response}

    raise ValueError("Reponse vide.")
